$First test input file
$this is a valid program, and should pass parsing with zero errors

begin
	const n = 54; integer array A[n];
	integer x, i; Boolean found;

proc Search
begin integer m;
      i, m := 1, n;
      do i < m ->
      	if A[i] = x -> m := i; []
	 ~(A[i]=x)-> i := i + 1;
	fi;
      od;
      found := A[i] = x;
      end;

$input table
i := 1;
do ~(i>n)-> read A[i]; i:=i+1; od;

read i;
do ~(x=0)->
   call Search;
   if found -> write x, i; []
   ~found -> write x;
   fi;
   read x;
od;
end.
